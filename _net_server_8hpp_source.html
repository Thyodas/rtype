66 void dispatchResponse(const std::shared_ptr<rtype::net::Connection<common::NetworkMessage>>& client, const rtype::net::Message<common::NetworkMessage>& msg)
67 {
68 std::vector<ResponseFunction> functionsToCall;
69
70 {
71 const auto &[first, second] = _responses.equal_range(msg.header.id);
72 for (auto it = first; it != second; ++it) {
73 functionsToCall.push_back(it->second);
74 }
75 }
76
77 for (auto& func : functionsToCall) {
78 func(client, msg);
79 }
80 }
