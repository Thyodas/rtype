70 friend Message<T>& operator<<(Message<T>& msg, const DataType& data)
71 {
72 // Create a stream
73 std::stringstream ss;
74
75 // Create a boost archive from the stream
76 boost::archive::binary_oarchive oa(ss);
77
78 // Serialize the data into the archive
79 oa << data;
80
81 // Convert the serialized data to a vector of bytes and append it to the message body
82 std::string s = ss.str();
83 std::vector<uint8_t> bytes(s.begin(), s.end());
84 msg.body.insert(msg.body.end(), bytes.begin(), bytes.end());
85
86 // Update the message header size
87 msg.header.size = msg.body.size();
88
89 return msg;
90 }

101 {
102 // Create a stream from the message body

104 std::stringstream ss(s);
105
106 // Create a boost archive from the stream

108
109 try {
110 // Deserialize the data from the archive

113 // TODO: find a better way of handling this error
