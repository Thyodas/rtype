158 template<typename T>
159 void ServerInterface<T>::waitForClientConnection()
160 {
161 m_asioSocket.async_receive_from(
162 asio::buffer(&_msgTemporaryIn, sizeof(Message<T>)), remoteEndpoint,
163 [this](std::error_code ec, std::size_t length) {
164 if (ec) {
165 std::cout << "[SERVER] Receive Error: " << ec.message() << "\n";
166 waitForClientConnection();
167 return;
168 }
169
170 std::cout << "[SERVER] New Connection: " << remoteEndpoint << "\n";
171
172 std::shared_ptr<Connection<T>> newconn =
173 std::make_shared<Connection<T>>(Connection<T>::owner::server,
174 m_asioContext, m_asioSocket, m_qMessagesIn);
175
176 if (onClientConnect(newconn)) {
177 m_deqConnections.push_back(std::move(newconn));
178 m_mapConnections.emplace(nIDCounter, m_deqConnections.back());
179
180 m_deqConnections.back()->connectToClient(nIDCounter);
181
182 nIDCounter++;
183
184 std::cout << "[" << m_deqConnections.back()->getID() << "] Connection Approved\n";
185 } else {
186 std::cout << "[-----] Connection Denied\n";
187 }
188 waitForClientConnection();
189 });
190 }

194 {
195 if (!client || !client->isConnected()) {
196 onClientDisconnect(client);

203 return;

209 void ServerInterface<T>::messageAllClients(const Message<T>& msg, std::shared_ptr<Connection<T>> pIgnoreClient)
210 {
211 bool bInvalidClientExists = false;
212
213 for (auto& client: m_deqConnections) {
214 if (!client || !client->isConnected()) {
215 onClientDisconnect(client);
216
217 if (client)
218 m_mapConnections.erase(client->getID());
219 client.reset();
220
221 bInvalidClientExists = true;
222 continue;
223 }
224 if (client != pIgnoreClient)
225 client->send(msg);
226 }
227
228 if (bInvalidClientExists) {
229 m_deqConnections.erase(
230 std::remove(m_deqConnections.begin(),
231 m_deqConnections.end(),
232 nullptr),
233 m_deqConnections.end());
234 }

237 template<typename T>
238 void ServerInterface<T>::update(size_t nMaxMessages, bool bWait)
239 {
240 if (bWait)
241 m_qMessagesIn.wait();
242 size_t nMessageCount = 0;
243 while (nMessageCount < nMaxMessages && !m_qMessagesIn.empty()) {
244 auto ownedMessage = m_qMessagesIn.pop_front();
245 onMessage(ownedMessage.remote, ownedMessage.msg);
246 nMessageCount++;
247 }
