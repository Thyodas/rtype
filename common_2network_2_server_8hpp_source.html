147 template<typename T>
148 void ServerInterface<T>::waitForClientConnection()
149 {
150 m_asioAcceptor.async_accept(
151 [this](std::error_code ec, asio::ip::tcp::socket socket) {
152 if (ec) {
153 std::cout << "[SERVER] New Connection Error: " << ec.message() << "\n";
154 waitForClientConnection();
155 return;
156 }
157 std::cout << "[SERVER] New Connection: " << socket.remote_endpoint() << "\n";
158
159 std::shared_ptr<Connection<T>> newconn =
160 std::make_shared<Connection<T>>(Connection<T>::owner::server,
161 m_asioContext, std::move(socket), m_qMessagesIn);
162
163 if (onClientConnect(newconn)) {
164 m_deqConnections.push_back(std::move(newconn));
165 m_mapConnections.emplace(nIDCounter, m_deqConnections.back());
166
167 m_deqConnections.back()->connectToClient(nIDCounter);
168
169 nIDCounter++;
170
171 std::cout << "[" << m_deqConnections.back()->getID() << "] Connection Approved\n";
172 } else {
173 std::cout << "[-----] Connection Denied\n";
174 }
175 waitForClientConnection();
176 });
177 }

180 void ServerInterface<T>::messageClient(std::shared_ptr<Connection<T>> client, const Message<T>& msg)
181 {
182 if (!client || !client->isConnected()) {
183 onClientDisconnect(client);
184
185 if (client)
186 m_mapConnections.erase(client->getID());
187 client.reset();
188 m_deqConnections.erase(
189 std::remove(m_deqConnections.begin(), m_deqConnections.end(), client), m_deqConnections.end());
190 return;
191 }
192 client->send(msg);
193 }

195 template<typename T>
196 void ServerInterface<T>::messageAllClients(const Message<T>& msg, std::shared_ptr<Connection<T>> pIgnoreClient)
197 {
198 bool bInvalidClientExists = false;
199
200 for (auto& client: m_deqConnections) {
201 if (!client || !client->isConnected()) {
202 onClientDisconnect(client);
203
204 if (client)
205 m_mapConnections.erase(client->getID());
206 client.reset();
207
208 bInvalidClientExists = true;
209 continue;
210 }
211 if (client != pIgnoreClient)
212 client->send(msg);
213 }
214
215 if (bInvalidClientExists) {
216 m_deqConnections.erase(
217 std::remove(m_deqConnections.begin(),
218 m_deqConnections.end(),
219 nullptr),
220 m_deqConnections.end());
221 }

226 {
227 if (bWait) m_qMessagesIn.wait();

229 size_t nMessageCount = 0;
230 while (nMessageCount < nMaxMessages && !m_qMessagesIn.empty()) {
231 auto msg = m_qMessagesIn.pop_front();

233 onMessage(msg.remote, msg.msg);
234
235 nMessageCount++;
