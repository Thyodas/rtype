127 std::vector<int> registerResponse(const std::vector<std::pair<common::NetworkMessage, ResponseFunction>>& responses)
128 {
129 std::vector<int> registeredIds;
130 for (const auto& response : responses) {
131 int id = _nextId++;
132 _responses.insert(ResponseRecord(id, response.first, response.second));
133 registeredIds.push_back(id);
134 }
135 return registeredIds;
136 }

138 void unregisterResponse(int responseId)
139 {
140 auto& id_index = _responses.get<id>();
141 id_index.erase(responseId);
142 }

144 void dispatchResponse(rtype::net::Message<common::NetworkMessage>& msg)
145 {
146 std::vector<ResponseFunction> functionsToCall;
147
148 auto& message_type_index = _responses.get<message_type>();
149 auto range = message_type_index.equal_range(msg.header.id);
150
151 for (auto it = range.first; it != range.second; ++it) {
152 functionsToCall.push_back(it->func);
153 }
154
155 for (auto& func : functionsToCall) {
156 func(msg);
157 }
158 }

160 void dispatchAllResponses()
161 {
162 auto &incoming = getIncoming();
163 while (!incoming.empty()) {
164 auto msg = incoming.pop_front().msg;
165 dispatchResponse(msg);
166 }
