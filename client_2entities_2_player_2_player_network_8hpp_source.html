135 Vector3 calculateBulletVelocity(const ecs::components::physics::transform_t& shipTransform, float bulletSpeed)
136 {
137 // Assuming the ship's forward direction is the negative z-axis
138 Vector3 shipForward = shipTransform.rotation;
139
140 // Negate the z-axis to get the forward direction
141 shipForward.z = -shipForward.z;
142
143 // Calculate the bullet velocity
144 Vector3 bulletVelocity = Vector3Scale(shipForward, bulletSpeed);
145
