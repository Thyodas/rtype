79 bool ClientInterface<T>::connect(const std::string& host, const uint16_t port)
80 {
81 try {
82 asio::ip::tcp::resolver resolver(m_context);
83 asio::ip::tcp::resolver::results_type endpoints = resolver.resolve(host, std::to_string(port));
84
85 m_connection = std::make_unique<Connection<T>>(Connection<T>::owner::client, m_context,
86 asio::ip::tcp::socket(m_context), m_qMessagesIn);
87
88 m_connection->connectToServer(endpoints);
89
90 thrContext = std::thread([this]() {
91 m_context.run();
92 });
93 } catch (std::exception& e) {
94 std::cerr << "Client Exception: " << e.what() << "\n";
95 return false;
96 }
97 return true;
